\section{O que é Beamer e por que usá-lo?}


\begin{frame}[label=myslide01] % [<options>=label, plain, (c=center,t=top,b=bottom)]

\frametitle{O que é Beamer?} {\ % <= Be careful with this!
\begin{itemize}
    \item Beamer é uma classe de documento em \LaTeX{} que produz apresentações acadêmicas limpas, organizadas e padronizadas.
             \pause \item A maioria dos editores de \LaTeX{} necessita a instalação do pacote \emph{beamer}, que pode ser facilmente obtido da internet.
\end{itemize}

}  % <= Be careful with this!
\end{frame}

\begin{frame}[label=myslide02] % [<options>=label, plain, (c=center,t=top,b=bottom)]

\frametitle{Por que usá-lo?} {\ % <= Be careful with this!
\begin{itemize}
\item A maioria dos artigos acadêmicos em áreas que usam matemática em algum grau usam Beamer.
    \pause \item Muito conveniente, pois utiliza o mesmo código do seu artigo em \LaTeX{}.
    \pause \item Como todo documento em \LaTeX{}, permite controle global \textcolor[rgb]{1.00,0.00,0.00}{e} local de \emph{layout}, cor e fontes. Não importa se você pula uma linha ou pressiona a tecla de espaço mais de uma vez por acidente, nada muda no documento final.
         \pause \item O resultado final é um arquivo em PDF bastante portátil que se mantém inalterado em diferentes configurações e sistemas operacionais.
\end{itemize}

}  % <= Be careful with this!
\end{frame}


\section{Use Beamer em 4 passos}

\begin{frame}[label=usebeamer]
\frametitle{Use Beamer em 4 passos} {\
\transdissolve %For more options, see Beamer User Guide.
\vskip-0.35cm
\begin{enumerate}
  \item Instale um editor de \LaTeX{}.
  \item Instale Mik\TeX{}: Provê um conjunto de ferramentas necessárias para prepare documentos usando a linguagem \TeX{}/\LaTeX{}. (Este software automaticamente se acopla ao editor de \LaTeX{})
  \item Use Mik\TeX{} para fazer o download e instalação do pacote \emph{beamer}. (Se a versão do Mik\TeX{} for mais recente, quando o comando ``usepackage" for utilizado pela primeira vez, o pacote é automaticamente baixado e instalado.)
  \item Use o arquivo .tex da presente apresentação como modelo e siga os passos a seguir. (Use ``Um Tutorial de Latex usando Latex" como guia auxiliar.)
\end{enumerate}

}
\end{frame}

\subsection{Equações}

\begin{frame}[label=myslide02]\footnotesize
\frametitle{Equações} {\


\begin{itemize}
    \item Como em qualquer documento em \LaTeX{}, é muito fácil escrever equações:
\end{itemize}

\begin{equation}\label{eq:demand}
    Q =\alpha+\beta P+ \gamma I+\epsilon
\end{equation}

\begin{equation}\label{eq:sum}
   1 + 2 + \cdots + n = \frac{n(n + 1)}{2}
\end{equation}

}
\end{frame}

\begin{frame}[label=myslide02b, fragile]\footnotesize
\frametitle{Equações} {\

\vskip-0.3cm

\begin{itemize}
    \item As equações anteriores aparecerão se digitar:
\end{itemize}

\begin{verbatim}
\begin{equation}
    Q =\alpha+\beta P+ \gamma I+\epsilon
\end{equation}

\begin{equation}
   1 + 2 + \cdots + n = \frac{n(n + 1)}{2}
\end{equation}
\end{verbatim}

\begin{itemize}
    \item Não se assuste com a quantidade de códigos. A maioria dos editores de \LaTeX{} tem botões que ao serem clicados escrevem o código da letra grega ou notação matemática que precisa.
    \item Um recurso bastante útil para equções mais complicadas, mas frequentemente utilizadas é a \textcolor[rgb]{0.00,0.50,1.00}{\emph{Wikipedia}}. Ela tem todas a maioria das equações escritas em código de \LaTeX{}. Apenas procure pela fonte geradora da página correpondente.
\end{itemize}

}
\end{frame}

\begin{frame}[label=appendix3b, plain, fragile]\small
\frametitle{Blocos e Colunas} {\

Você pode combinar ambos os comandos:

\begin{verbatim}
\begin{columns}
\begin{column}{2.8cm}
\begin{block}{Smaller Column 1}Some text here.\end{block}
\end{column}
\begin{column}{5.5cm}
\begin{block}{Bigger Column 2}Maybe something else here.\end{block}
\end{column}
\end{columns}
\end{verbatim}

Vai aparecer:

\begin{columns}
\begin{column}{3cm}
\begin{block}{Coluna Pequena 1}Um pouco de texto aqui.\end{block}
\end{column}
\begin{column}{5.5cm}
\begin{block}{Coluna Grande 2}talvez um pouco mais de texto aqui.\end{block}

\end{columns}
